//Решить задачу Цех по ремонту бытовой техники имеет ежедневно прибыль Р, с которой платит налог 70%.
//Из оставшейся доли прибыли 30% идет на развитие производства, а остальное – на заработную плату 8 работникам. Какова среднемесячная зарплата одного работника?
#include <iostream>
#include <stdio.h>
#include <math.h>

using namespace std;
int main() 
{
    int p, a;
    cout << "Введите ежедневную прибыль p = ";
    cin >> p;
    a = (((0.3 * p) * 0.7) / 8) * 30;
    cout << "Среднемесячная зарплата одного работника = " << a;
    return 0; 
}
